    <script>
        // Cargar Google Charts
        google.charts.load('current', { packages: ['corechart'] });

        // Dibujar el Gráfico Mensual
        google.charts.setOnLoadCallback(drawMonthlyChart);
        function drawMonthlyChart() {
            var data = google.visualization.arrayToDataTable([
                ['Mes', 'Gastos'],
                ['Ene', 3000],
                ['Feb', 1500],
                ['Mar', 2000],
                ['Abr', 2500],
                ['May', 2800],
                ['Jun', 1800],
                ['Jul', 1000],
                ['Ago', 2200],
                ['Sep', 2700],
                ['Oct', 2900],
                ['Nov', 3200],
                ['Dic', 1500]
            ]);

            var options = {
                title: '',
                legend: { position: 'none' },
                colors: ['#007bff']
            };

            var chart = new google.visualization.ColumnChart(document.getElementById('monthlyChart'));
            chart.draw(data, options);
        }

        // Dibujar el Gráfico por Categoría
        google.charts.setOnLoadCallback(drawCategoryChart);
        function drawCategoryChart() {
            var data = google.visualization.arrayToDataTable([
                ['Categoría', 'Gastos'],
                ['Servicios', 4000],
                ['Ocio', 2000],
                ['Alimentación', 3000]
            ]);

            var options = {
                title: '',
                legend: { position: 'none' },
                colors: ['#007bff', '#6c757d', '#ffc107']
            };

            var chart = new google.visualization.BarChart(document.getElementById('categoryChart'));
            chart.draw(data, options);
        }
    </script>
